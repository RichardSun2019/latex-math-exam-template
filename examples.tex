    \begin{cases}
    (1) & \bm{a} \cdot \bm{b} = \bm{b} \cdot \bm{a}, \\[0.3em]
    (2) & (\lambda \bm{a}) \cdot \bm{b} = \lambda (\bm{a} \cdot \bm{b}) = \bm{a} \cdot (\lambda \bm{b}), \\[0.3em]
    (3) & (\bm{a} + \bm{b}) \cdot \bm{c} = \bm{a} \cdot \bm{c} + \bm{b} \cdot \bm{c}.
    \end{cases}
    \]\vspace{15em}
